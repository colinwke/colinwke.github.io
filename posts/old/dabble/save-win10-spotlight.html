@echo off
MD wallpaper
xcopy "%UserProfile%\AppData\Local\Packages\Microsoft.Windows.ContentDeliveryManager_cw5n1h2txyewy\LocalState\Assets" "D:\spotlight\" /S /Y /D

D:
cd "D:\spotlight\"
ren * *.jpg
pause

2 双击执行, 执行后前往D:\spotlight\查看
